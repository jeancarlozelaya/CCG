Crear mapa de ID -> nombre actual (de las nuevas categorías) var mapaIdANombre = {}; var mapaNombreAUrl = {}; for (var i = 0; i < categoriasNuevas.length; i++) { for (var j = 0; j < categoriasNuevas[i].productos.length; j++) { var prod = categoriasNuevas[i].productos[j]; if (prod.id) { mapaIdANombre[prod.id] = prod.nombre; if (prod.imagen && prod.imagen !== "" && !prod.imagen.startsWith('__PENDIENTE__')) { mapaNombreAUrl[prod.nombre] = prod.imagen; } } } } // 2. Crear mapa de ID -> nombre original (de las categorías originales) var mapaIdANombreOriginal = {}; for (var i = 0; i < categoriasOriginales.length; i++) { for (var j = 0; j < categoriasOriginales[i].productos.length; j++) { var prod = categoriasOriginales[i].productos[j]; if (prod.id) { mapaIdANombreOriginal[prod.id] = prod.nombre; } } } // 3. Obtener encabezados actuales de Requisiciones (fila 1) var ultimaColumna = requisicionesSheet.getLastColumn(); var encabezadosActuales = []; if (ultimaColumna >= 5) { encabezadosActuales = requisicionesSheet.getRange(1, 5, 1, ultimaColumna - 4).getValues()[0]; } Logger.log("Encabezados actuales: " + JSON.stringify(encabezadosActuales)); // 4. Crear mapa de nombre actual en Requisiciones -> columna var mapaNombreExistente = {}; for (var i = 0; i < encabezadosActuales.length; i++) { if (encabezadosActuales[i] && encabezadosActuales[i] !== "") { mapaNombreExistente[encabezadosActuales[i]] = i; } } // 5. Construir nuevos encabezados y fórmulas de imagen var nuevosEncabezados = []; var nuevasFormulasImagen = []; var nombresProcesados = new Set(); // Primero, procesar productos existentes por ID var idsProcesados = new Set(); for (var i = 0; i < encabezadosActuales.length; i++) { var nombreActual = encabezadosActuales[i]; if (nombreActual && nombreActual !== "") { // Buscar a qué ID pertenece este nombre en las categorías originales var idEncontrado = null; for (var id in mapaIdANombreOriginal) { if (mapaIdANombreOriginal[id] === nombreActual) { idEncontrado = id; break; } } // Si encontramos el ID, verificar si el producto aún existe y su nuevo nombre if (idEncontrado && mapaIdANombre[idEncontrado]) { var nuevoNombre = mapaIdANombre[idEncontrado]; nuevosEncabezados.push(nuevoNombre); nombresProcesados.add(nuevoNombre); idsProcesados.add(idEncontrado); // Agregar fórmula de imagen var urlImagen = mapaNombreAUrl[nuevoNombre] || ""; nuevasFormulasImagen.push(urlImagen ? '=IMAGE("' + urlImagen + '")' : ''); Logger.log("Producto existente: " + nombreActual + " -> " + nuevoNombre + " (ID: " + idEncontrado + ")"); } else { // Producto que ya no existe en el nuevo catálogo (fue eliminado) // Mantenemos la columna pero sin fórmula de imagen nuevosEncabezados.push(nombreActual); nuevasFormulasImagen.push(''); Logger.log("Producto eliminado (se mantiene columna): " + nombreActual); } } } // 6. Agregar productos NUEVOS (que tienen ID pero no estaban en los encabezados) for (var i = 0; i < categoriasNuevas.length; i++) { for (var j = 0; j < categoriasNuevas[i].productos.length; j++) { var prod = categoriasNuevas[i].productos[j]; if (prod.id && !idsProcesados.has(prod.id) && !nombresProcesados.has(prod.nombre)) { nuevosEncabezados.push(prod.nombre); nombresProcesados.add(prod.nombre); var urlImagen = (prod.imagen && !prod.imagen.startsWith('__PENDIENTE__')) ? prod.imagen : ""; nuevasFormulasImagen.push(urlImagen ? '=IMAGE("' + urlImagen + '")' : ''); Logger.log("Producto NUEVO agregado al final: " + prod.nombre + " (ID: " + prod.id + ")"); } } } Logger.log("Nuevos encabezados finales: " + JSON.stringify(nuevosEncabezados)); // 7. Escribir los encabezados y fórmulas de imagen if (nuevosEncabezados.length > 0) { // Limpiar encabezados existentes if (ultimaColumna >= 5) { requisicionesSheet.getRange(1, 5, 1, ultimaColumna - 4).clearContent(); if (requisicionesSheet.getLastRow() >= 2) { requisicionesSheet.getRange(2, 5, 1, ultimaColumna - 4).clearContent(); } } // Escribir nuevos encabezados en fila 1 var rangoEncabezados = requisicionesSheet.getRange(1, 5, 1, nuevosEncabezados.length); rangoEncabezados.setValues([nuevosEncabezados]); rangoEncabezados.setFontWeight("bold"); rangoEncabezados.setBackground("#2c3e50"); rangoEncabezados.setFontColor("white"); // Escribir fórmulas de imagen en fila 2 if (nuevasFormulasImagen.length > 0) { var rangoImagenes = requisicionesSheet.getRange(2, 5, 1, nuevasFormulasImagen.length); rangoImagenes.setValues([nuevasFormulasImagen]); requisicionesSheet.setRowHeight(2, 80); } Logger.log("Encabezados actualizados. Total: " + nuevosEncabezados.length); } } catch (e) { Logger.log("Error actualizando Requisiciones: " + e.message); Logger.log("Stack: " + e.stack); } }
